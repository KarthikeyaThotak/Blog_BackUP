# Project:  Finding Missing Number in an list


# Function: that can find missing numbers

def findMissingNumbers(n):
    number = set(n)
    length = len(n)
    output = []
    for i in range(1, n[-1]):
        if i not in number:
            output.append(i)
    return output

number_list = [1,3,4,5,7,8,9,11,12,18,20]
print(findMissingNumbers(number_list))
