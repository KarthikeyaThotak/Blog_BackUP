from collections import defaultdict

# Making an function to group the words which are anagrams of each other.

def anagrams(words):
    df = defaultdict(list) #DefaultDict needs to take an datatype, in this case we are using an list
    for i in words:
        sort = " ".join(sorted(i))
        print(sort)
        df[sort].append(i)
        print(sort)
    print(sort)
    return df.values()
# TestCase
words = ["race", "care", "part", "trap", "heart", "earth", "Listen", "Silent"]
print(anagrams(words))
